Lisp  -- symbolic -- atoms(symbols) and lists

(defun fibonacci (n &optional (a 0) (b 1) (acc ()))
  (if (zerop n)
      (nreverse acc)
      (fibonacci (1- n) b (+ a b) (cons a acc))))

(fibonacci 5) ; ==> (0 1 1 2 3)

// fibonacci reclusive
function fib(n) {
  if (n <= 2) return 1;
  return fib(n-2) + fib(n-1);
}


// fibonacci iteration
function fib_iter(num) {
  var a = 1, b = 0, temp;
  while (num >= 0){
   temp = a;
   a = a + b;
   b = temp;
   num--;
  }
  return b;
}
